git init    # когда берём новую папку и просим чтобы GIT следил за изменениями в этой папке
git add     # перед тем как сделать Commit мы пишем все вносим изменения в папке
git status  # мы смотрим всё ли ОК
git commit  # мы сохраняем всё внесенное в GIT
ESC + :x! (после внесения комментарий при git commit)
git log     # смотрим все внесённые изменения с предоставлением кода
git log --graph  # смотрим все внесённые изменения с предоставлением кода более красочно
